<script>
const maxHand = 5;
let hand = [];
let aiHand = [];

function drawCard() {
  if (hand.length >= maxHand) {
    alert("手牌已滿！");
    return;
  }
  const card = Math.floor(Math.random() * 10) + 1;
  hand.push(card);
  updateHands();
}

function updateHands() {
  const handDiv = document.getElementById("hand");
  handDiv.innerHTML = "";
  hand.forEach((c, i) => {
    const cardDiv = document.createElement("div");
    cardDiv.className = "card";
    cardDiv.textContent = c;
    cardDiv.onclick = () => playCard(i);
    handDiv.appendChild(cardDiv);
  });
  document.getElementById("hand-count").textContent = hand.length;

  // AI 隨機手牌顯示
  while(aiHand.length < 5) {
    aiHand.push(Math.floor(Math.random() * 10) + 1);
  }
  const aiDiv = document.getElementById("ai-hand");
  aiDiv.innerHTML = "";
  aiHand.forEach(c => {
    const cardDiv = document.createElement("div");
    cardDiv.className = "card";
    cardDiv.textContent = "?";
    aiDiv.appendChild(cardDiv);
  });
}

function playCard(index) {
  const playerCard = hand.splice(index, 1)[0];
  const aiCard = aiHand.shift();
  const status = document.getElementById("status");
  if (playerCard > aiCard) status.textContent = `你出 ${playerCard}，AI 出 ${aiCard}，你贏了！`;
  else if (playerCard < aiCard) status.textContent = `你出 ${playerCard}，AI 出 ${aiCard}，你輸了！`;
  else status.textContent = `你出 ${playerCard}，AI 出 ${aiCard}，平手！`;
  updateHands();
}

// 初始化
updateHands();
document.getElementById("draw-btn").onclick = drawCard;
</script>
