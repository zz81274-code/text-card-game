<script>
const maxHand = 5;
let hand = [];
let aiHand = [];
let timer = 15;
let timerInterval;

function drawCard() {
  if (hand.length >= maxHand) { alert("手牌已滿！"); return; }
  const card = Math.floor(Math.random() * 10) + 1;
  hand.push(card);
  updateHands();
}

function updateHands() {
  const handDiv = document.getElementById("hand");
  handDiv.innerHTML = "";
  hand.forEach((c, i) => {
    const cardDiv = document.createElement("div");
    cardDiv.className = "card";
    cardDiv.textContent = c;
    cardDiv.onclick = () => playCard(i);
    handDiv.appendChild(cardDiv);
  });
  document.getElementById("hand-count").textContent = hand.length;

  while(aiHand.length < 5) { aiHand.push(Math.floor(Math.random() * 10) + 1); }
  const aiDiv = document.getElementById("ai-hand");
  aiDiv.innerHTML = "";
  aiHand.forEach(c => {
    const cardDiv = document.createElement("div");
    cardDiv.className = "card";
    cardDiv.textContent = "?";
    aiDiv.appendChild(cardDiv);
  });
}

function startTurn() {
  clearInterval(timerInterval);
  timer = 15;
  document.getElementById("timer").textContent = "回合倒數: 15 秒";
  timerInterval = setInterval(() => {
    timer--;
    document.getElementById("timer").textContent = "回合倒數: " + timer + " 秒";
    if(timer <= 0) {
      clearInterval(timerInterval);
      autoPlay();
    }
  }, 1000);
}

function playCard(index) {
  clearInterval(timerInterval);
  const playerCardDiv = document.getElementById("hand").children[index];
  playerCardDiv.classList.add("played");

  const playerCard = hand.splice(index, 1)[0];
  const aiCard = aiHand.shift();
  const aiDiv = document.getElementById("ai-hand").children[0];
  aiDiv.classList.add("played");

  const status = document.getElementById("status");
  setTimeout(() => {
    if(playerCard > aiCard) status.textContent = `你出 ${playerCard}，AI 出 ${aiCard}，你贏了！`;
    else if(playerCard < aiCard) status.textContent = `你出 ${playerCard}，AI 出 ${aiCard}，你輸了！`;
    else status.textContent = `你出 ${playerCard}，AI 出 ${aiCard}，平手！`;
    updateHands();
    startTurn();
  }, 500);
}

function autoPlay() {
  if(hand.length > 0) playCard(0);
}

document.getElementById("draw-btn").onclick = drawCard;

updateHands();
startTurn();
</script>
